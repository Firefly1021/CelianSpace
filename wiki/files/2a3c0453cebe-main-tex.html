两个求和公式及应用

Euler-Maclaurin求和公式

级数的一个重要问题在于如何判断其收敛性,这一问题本质上就是分析级数的渐近性质,也就是怎么增长或衰减的.这在数论中(尤其是解析数论里)十分重要.下面给出一般的Euler-Maclaurin求和公式,它是级数收敛性判断的一个重要工具.

\[\sum_{n=a}^{b}f(n)=\int_{a}^{b}f(x)dx+\frac{f(a)+f(b)}{2}+\sum_{k=1}^{\lfloor m/2\rfloor }\frac{B_{2k}}{(2k)!}(f^{(2k-1)}(b)-f^{(2k-1)}(a))+R_m\]
\[B_0(x)=1, B_n'(x)=nB_{n-1}(x), \int_0^1 B_n(x) dx=0\]
\[\begin{aligned} B_0(x) &= 1 \\ B_1(x) &= x - \frac{1}{2} \\ B_2(x) &= x^2 - x + \frac{1}{6} \\ B_3(x) &= x^3 - \frac{3}{2}x^2 + \frac{1}{2}x \\ B_4(x) &= x^4 - 2x^3 + x^2 - \frac{1}{30} \\ \end{aligned}\]
\[B_0=1, B_1=-\frac{1}{2}, B_2=\frac{1}{6}, B_3=0, B_4=-\frac{1}{30}, B_5=0, B_6=\frac{1}{42}, \cdots\]
\[\sum_{n=a}^{b}f(n)=\int_{a}^{b}f(x)dx+\frac{f(a)+f(b)}{2}+\int_{a}^{b}(x-\lfloor x \rfloor-\frac{1}{2})f'(x)dx\]
\[\begin{aligned} \int_{a}^{b}(x-\lfloor x \rfloor-\frac{1}{2})f'(x)dx &= \sum_{k=a}^{b-1}\int_{k}^{k+1}(x-\lfloor x \rfloor-\frac{1}{2})f'(x)dx \\ &= \sum_{k=a}^{b-1}\int_{0}^{1}(x-\frac{1}{2})f'(k+x)dx \\ &= \sum_{k=a}^{b-1}\left[\frac{f(k)+f(k+1)}{2}-\int_{k}^{k+1}f(x)dx\right] \\ &= \sum_{k=a}^{b}f(k)-\int_{a}^{b}f(x)dx-\frac{f(a)+f(b)}{2} \end{aligned}\]
\[\sum_{n=1}^{N}\frac{1}{n} = \ln N+\frac{1}{2}+\frac{1}{2N}-\int_{1}^{N}\frac{x-\lfloor x \rfloor-1/2}{x^2}dx\]
\[\sum_{n=1}^{N}\frac{\ln n}{n^2+1} = \int_{1}^{N}\frac{\ln x}{x^2+1}dx+\frac{\ln N}{2(N^2+1)}+\int_{1}^{N}(x-\lfloor x \rfloor-\frac{1}{2})\left(\frac{1+x^2(1-2\ln x)}{x(x^2+1)^2}\right)dx\]
\[\int_{1}^{\infty}\frac{\ln x}{x^2+1}dx, \int_{1}^{\infty}(x-\lfloor x \rfloor-\frac{1}{2})\left(\frac{1+x^2(1-2\ln x)}{x(x^2+1)^2}\right)dx\]
\[\begin{aligned} \ln N! = \sum_{n=1}^{N}\ln n &= \int_{1}^{N}\ln x dx + \frac{\ln N}{2} + \int_{1}^{N}(x-\lfloor x \rfloor-\frac{1}{2})\frac{1}{x}dx\\ &= N\ln N - N + 1 + \frac{\ln N}{2} + \int_{1}^{N}(x-\lfloor x \rfloor-\frac{1}{2})\frac{1}{x}dx \end{aligned}\]
\[\begin{aligned} \int_{1}^{N}(x-\lfloor x \rfloor-\frac{1}{2})\frac{1}{x}dx &= \frac{1}{2}\int_1^N \frac{1}{x}\mathrm{d}B_2(x-\lfloor x \rfloor) \\ &= \frac{1}{2}\left\{\left[\frac{B_2(x-\lfloor x \rfloor)}{x}\right]_{1}^{N}-\int_1^N B_2(x-\lfloor x \rfloor)\mathrm{d}\left(\frac{1}{x}\right)\right\} \\ \end{aligned}\]
\[\ln N!= N\ln N - N + \frac{\ln N}{2} +\left(1-\frac{1}{12}\right)+\frac{1}{12N} + \frac{1}{2}\int_1^N B_2(x-\lfloor x \rfloor)\frac{1}{x^2}dx\]
\[\int_1^N B_2(x-\lfloor x \rfloor)\frac{1}{x^2}dx=\int_1^\infty B_2(x-\lfloor x \rfloor)\frac{1}{x^2}dx - \int_N^\infty B_2(x-\lfloor x \rfloor)\frac{1}{x^2}dx\]
\[\left|\int_N^\infty B_2(x-\lfloor x \rfloor)\frac{1}{x^2}dx\right| \leqslant \frac{1}{6} \int_N^\infty \frac{1}{x^2}dx = \frac{1}{6N}\]
\[\ln N! = N\ln N - N + \frac{\ln N}{2} + C + O\left(\frac{1}{N}\right)\]

Abel求和公式

\[\sum_{1\leqslant n\leqslant x}a_nf(n) = A(x)f(x) - \int_1^x A(t)f'(t)dt\]
\[\begin{aligned} \sum_{1\leqslant n\leqslant x}a_nf(n) &= \sum_{1\leqslant n\leqslant k}a_nf(n)\\ &= A(k)f(k)-\sum_{1\leqslant n\leqslant k-1}A(n)(f(n+1)-f(n)) \\ \end{aligned}\]
\[\begin{aligned} \sum_{1\leqslant n\leqslant k-1}A(n)(f(n+1)-f(n)) &= \sum_{1\leqslant n\leqslant k-1}A(n)\int_n^{n+1}f'(t)dt \\ &= \sum_{1\leqslant n\leqslant k-1}\int_n^{n+1}A(t)f'(t)dt\\ &= \int_1^k A(t)f'(t)dt=\int_1^x A(t)f'(t)dt-\int_k^x A(t)f'(t)dt \end{aligned}\]
\[\int_k^x A(t)f'(t)dt= A(k)\int_k^x f'(t)dt=A(x)f(x)-A(k)f(k)\]
\[\sum_{1\leqslant n\leqslant x}a_nf(n) = A(x)f(x) - \int_1^x A(t)f'(t)dt\]
\[a_n=\mathbf{1}_p(n)= \begin{cases} 1, & \text{如果 $n$ 是素数} \\ 0, & \text{其他} \end{cases}\]
\[\sum_{p\leqslant x}\frac{1}{p} = \frac{\pi(x)}{x} + \int_2^x \frac{\pi(t)}{t^2}dt\]
\[\frac{\pi(x)}{x}= \frac{1}{\ln x} + \frac{E(x)}{x}= \frac{1}{\ln x} + O\left(\frac{1}{\ln^2 x}\right)\]
\[\int_2^x \frac{\pi(t)}{t^2}dt= \int_2^x \frac{1}{t\ln t}dt + \int_2^x \frac{E(t)}{t^2}dt=\ln\ln x-\ln\ln 2 + \int_2^x \frac{E(t)}{t^2}dt\]
\[\int_2^{\infty}\left|\frac{E(t)}{t^2}\right|dt \leqslant C\int_2^{\infty} \frac{1}{t\ln^2 t}dt=\frac{C}{\ln 2}\]
\[\left|\int_x^{\infty}\frac{E(t)}{t^2}dt\right|\leqslant C\int_x^{\infty} \frac{1}{t\ln^2 t}dt = \frac{C}{\ln x}\]
\[\begin{aligned} \sum_{p\leqslant x}\frac{1}{p} &= \frac{\pi(x)}{x} + \int_2^x \frac{\pi(t)}{t^2}dt\\ &= \frac{1}{\ln x} + O\left(\frac{1}{\ln^2 x}\right) + \ln\ln x-\ln\ln 2 + C_1 + O\left(\frac{1}{\ln x}\right) \\ \end{aligned}\]
\[\sum_{p\leqslant x}\frac{1}{p} = \ln\ln x + M + O\left(\frac{1}{\ln x}\right)\]
