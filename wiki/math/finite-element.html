有限元方法初步

变分问题

设$a(\cdot,\cdot)$是赋范线性空间$H$上的双线性形式.若存在常数$C>0$使得

\[|a(v,w)|\leqslant C\|v\|_H\|w\|_H, \forall v,w\in H\]

我们称$a(\cdot,\cdot)$是有界的(连续的).若存在常数$\alpha >0$使得

\[a(v,v)\geqslant \alpha \|v\|^2_H, \forall v\in V\subset H\]

我们称$a(\cdot,\cdot)$在$V\subset H$上强制.

变分问题:设$H$为Hilbert空间,$V\subset H$为闭子空间,$a(\cdot,\cdot)$为$H$上的有界且在$V$上强制的双线性形式,那么非对称变分问题是指给定$F\in V'$,找到$u\in V$使得

Lax-Milgram定理
\[a(u,v)=\langle F,v\rangle, \forall v\in H\]
\[u_h=\sum_{j=1}^nU_j\phi_j\]
\[\begin{cases} U_1a(\phi_1,\phi_1)+U_2a(\phi_2,\phi_1)+\cdots+U_na(\phi_n,\phi_1)=\langle F,\phi_1\rangle\\ U_1a(\phi_1,\phi_2)+U_2a(\phi_2,\phi_2)+\cdots+U_na(\phi_n,\phi_2)=\langle F,\phi_2\rangle\\ \cdots\cdots\cdots\cdots\cdots\cdots\\ U_1a(\phi_1,\phi_n)+U_2a(\phi_2,\phi_n)+\cdots+U_na(\phi_n,\phi_n)=\langle F,\phi_n\rangle \end{cases}\]
\[KU=F\]
Céa引理
\[\|u-u_h\|_V\leqslant \frac{C}{\alpha}\inf_{v\in V_h}\|u-v\|_V\]
\[a(u-u_h,v)=0, \forall v\in V_h\]
\[\begin{aligned} \alpha \|u-u_h\|^2_V &\leqslant a(u-u_h,u-u_h)\\ &=a(u-u_h,u-v)+a(u-u_h,v-u_h)\\ &= a(u-u_h,u-v)\\ &\leqslant C\|u-u_h\|_V\|u-v\|_V \end{aligned}\]
\[\|u-u_h\|_V\leqslant \frac{C}{\alpha}\inf_{v\in V_h}\|u-v\|_V\]

有限元空间的构造

\[V_h=\{p\in \mathcal{P}_k(\Omega)\}\]
\[N_i(\phi_j)=\delta_{ij}= \begin{cases} 1,& i=j\\ 0,&i\neq j \end{cases}\]
\[0=x_0<x_1<\cdots<x_{n-1}<x_n=1\]
\[\mathcal{P}=\operatorname{span}\{1,x\}\]
\[\phi_{i-1}(x_{i-1})=1,\phi_{i}(x_{i-1})=0;\phi_{i-1}(x_i)=0,\phi_{i}(x_i)=1\]
\[N_{i-1}(\phi)=\phi(x_{i-1});N_{i}(\phi)=\phi(x_i)\]
\[N_i(\phi)=\phi(x_i)\]
\[\begin{aligned} N_1(\psi) &=\psi(0)=a=0\\ N_2(\psi) &=\psi(1)=a+b=0 \end{aligned}\]
\[\phi_1=1-x-y,\phi_2=x,\phi_3=y\]
\[\mathcal{I}_K f=\sum_{i=1}^{k}N_i(f)\phi_i\]
\[\mathcal{I}_{\mathcal{T}}f|_{K_i}=\mathcal{I}_{K_i} f\]
\[\mathcal{I}_{\mathcal{T}}f= \begin{cases} x+y,& (x,y)\in K_1\\ 2-x-y,& (x,y)\in K_2 \end{cases}\]

\[\sum\limits_{{j = 1}}^{n}{a}_{ij}\left( x\right) \frac{ tial p}{ tial {x}_{j}}\left( x\right) = {u}_{i}\left( x\right) \;\forall i = 1,\ldots ,n.\]
\[V:=\left\{ {v \in {L}^{2}{\left( \Omega \right) }^{n} : \operatorname{div}v \in {L}^{2}\left( \Omega \right) }\right\} .\]
\[a(v,v)\geqslant \alpha\|v\|^2_V\]

抽象混合公式

\[b(v,p)=(y_v,p), \forall p\in\varPi\]

\[a(u_0,v)=F(v)-a(u_1,v), \forall v\in Z\]
\[\alpha \| v\|_{V} \leq \mathop{\sup }\limits_{{w \in V}}\frac{a\left( {w,v}\right) }{\| w\|_{V}}\;\forall v \in V\]
\[\| {Tp}\|_{V} = \mathop{\sup }\limits_{{v \in {Z}^{ \bot }}}\frac{b\left( {v,p}\right) }{\| v\|_{V}} \leq C\| p\|_{\varPi },\]
\[\widetilde{F}\left( v\right) = {\left( u,v\right) }_{V}\;\forall v \in {Z}^{ \bot }\]
\[\begin{aligned} \beta {\begin{Vmatrix}{p}_{j} - {p}_{k}\end{Vmatrix}}_{\varPi } &\leq \mathop{\sup }\limits_{{w \in {Z}^{ \bot }}}\frac{b\left( {w,{p}_{j} - {p}_{k}}\right) }{\| w\|_{V}}\\ &= \mathop{\sup }\limits_{{w \in {Z}^{ \bot }}}\frac{{\left( w,T{p}_{j} - T{p}_{k}\right) }_{V}}{\| w\|_{V}}\\ &= {\begin{Vmatrix}T{p}_{j} - T{p}_{k}\end{Vmatrix}}_{V} \end{aligned}\]
\[\| w\|_{{H}^{1}{\left( \Omega \right) }^{n}} \leq \left( {1/\beta }\right) \| p\|_{{L}^{2}\left( \Omega \right) }\]
\[\beta \| p\|_{{L}^{2}\left( \Omega \right) } = \beta \frac{b\left( {w,p}\right) }{\| p\|_{{L}^{2}\left( \Omega \right) }} \leq \frac{b\left( {w,p}\right) }{\| w\|_{{H}^{1}{\left( \Omega \right) }^{n}}} \leq \sqrt{n}\frac{b\left( {w,p}\right) }{\| w\|_{H\left( \operatorname{div}\right) }},\]

离散混合公式

\[\|u - u_{h}\|_{V} \leq \left( 1 + \frac{C}{\alpha} \right) \inf_{v \in Z_{h}} \|u - v\|_{V} + \frac{1}{\alpha} \sup_{w \in Z_{h} \setminus \{0\}} \frac{|a(u - u_{h}, w)|}{\|w\|_{V}}\]
\[|b(w, p - q)| \leq C \|w\|_{V} \|p - q\|_{\varPi}.\]
\[|a(u - u_{h}, w)| \leq C \|w\|_{V} \inf_{q \in \varPi_{h}} \|p - q\|_{\varPi}.\]
\[\|u - u_{h}\|_{V} \leq \left( 1 + \frac{C}{\alpha} \right) \inf_{v \in Z_{h}} \|u - v\|_{V} + \frac{C}{\alpha} \inf_{q \in \varPi_{h}} \|p - q\|_{\varPi}\]

速度近似的收敛性结果

\[\inf_{q \in V_{h}^{k - 1}} \|p - q\|_{L^{2}(\Omega)} \leq C h^{s} \|p\|_{H^{s}(\Omega)}, 0 \leq s \leq k.\]
\[\|\psi - \psi_{h}\|_{H^{2}(\Omega)} \leq C h^{s} |\psi|_{H^{s + 2}(\Omega)}.\]
\[Z = \left\{ v \in H(\operatorname{div}) : \operatorname{div}\underset{\sim}{v} = 0 \right\}\]
\[V_{h} = V_{h}^{k} \times V_{h}^{k}\]
\[\|\underset{\sim}{a} \operatorname{grad}p - \underset{\sim}{u}_{h}\|_{L^{2}(\Omega)^{2}} \leq C h^{s + 1} \|\underset{\sim}{a} \operatorname{grad}p\|_{H^{s + 1}(\Omega)^{2}}\]
\[\|\underset{\sim}{u} - \underset{\sim}{u}_{h}\|_{H^{1}(\Omega)^{2}} \leq C h^{s} \|\underset{\sim}{u}\|_{H^{s + 1}(\Omega)^{2}}\]
\[\|u - u_{h}\|_{h} \leq C h \left( \|u\|_{H^{2}(\Omega)^{2}} + \|p\|_{H^{1}(\Omega)} \right).\]
\[\|\underset{\sim}{u} - u_{h}\|_{h} \leq \inf_{\underset{\sim}{v} \in Z_{h}} \|\underset{\sim}{u} - \underset{\sim}{v}\|_{h} + \sup_{\underset{\sim}{w} \in Z_{h} \setminus \{0\}} \frac{|a_{h}(\underset{\sim}{u} - u_{h}, \underline{w})|}{\|\underset{\sim}{w}\|_{h}}.\]
\[\begin{aligned} a_{h}(\underset{\sim}{u} - \underset{\sim}{u}_{h}, \underset{\sim}{w}) &= a_{h}(\underset{\sim}{u}, \underset{\sim}{w}) - \int_{\Omega} \underset{\sim}{f} \cdot \underset{\sim}{w} dx \\ &= a_{h}(\underset{\sim}{u}, \underset{\sim}{w}) - \int_{\Omega} \left( -\Delta \underset{\sim}{u} + \operatorname{grad}p \right) \cdot \underset{\sim}{w} dx &\text{(由 相关方程)} \\ &= \sum_{e} \int_{e} \sum_{i = 1,2} \frac{ tial u_{i}}{ tial \underset{\sim}{\nu}} [w_{i}] - p [\underset{\sim}{w} \cdot \underset{\sim}{\nu}] ds - b_{h}(\underset{\sim}{w}, p) \end{aligned}\]
\[\left| \sum_{e} \int_{e} \sum_{i = 1,2} \frac{ tial u_{i}}{ tial \underset{\sim}{\nu}} [w_{i}] - p [\underset{\sim}{w} \cdot \underset{\sim}{\nu}] ds \right| \leq C h \left( \|u\|_{H^{2}(\Omega)^{2}} + \|p\|_{H^{1}(\Omega)} \right) \|\underline{w}\|_{h}.\]
\[\begin{aligned} \left| b_{h}(w, p) \right| &\leq \sqrt{2} \|w\|_{h} \inf_{q \in \varPi_{h}} \|p - q\|_{L^{2}(\Omega)} \\ &\leq C h \|w\|_{h} \|p\|_{H^{1}(\Omega)} \end{aligned}\]
\[\inf_{v \in Z_{h}} \|u - v\|_{h} \leq C h \|u\|_{H^{2}(\Omega)^{2}}\]

离散 Inf-Sup 条件

\[T : {V}_{h} \rightarrow {\mathbb{R}}^{\dim {\varPi }_{h}}\text{ 其中 }{\left( Tv\right) }_{i}:=b\left( {v,{q}_{i}}\right)\]
\[\sum\limits_{{i = 1}}^{{\dim {\varPi }_{h}}}{c}_{i}{\left( Tv\right) }_{i} = 0\]
\[\dim {Z}_{h} = \dim \text{ kernel } T = \dim {V}_{h} - \dim \text{ image } T = \dim {V}_{h} - \dim {\varPi }_{h}.\]
\[\| M\|_{V \rightarrow \varPi }:=\mathop{\sup }\limits_{{0 \neq u \in {V}_{h}}}\frac{\| {Mu}\|_{\varPi }}{\| u\|_{V}} \leq \frac{1}{\beta }\]
\[{\begin{Vmatrix}p - {p}_{h}\end{Vmatrix}}_{\varPi } \leq \frac{C}{\beta }{\begin{Vmatrix}u - {u}_{h}\end{Vmatrix}}_{V} + \left( {1 + \frac{C}{\beta }}\right) \mathop{\inf }\limits_{{q \in {\varPi }_{h}}}\| p - q\|_{\varPi }\]
\[{\begin{Vmatrix}p - {p}_{h}\end{Vmatrix}}_{\varPi } \leq c\mathop{\inf }\limits_{{v \in {Z}_{h}}}\| u - v\|_{V} + \left( {1 + c}\right) \mathop{\inf }\limits_{{q \in {\varPi }_{h}}}\| p - q\|_{\varPi },\]
\[b\left( {w,q}\right) = b\left( {u - v,q}\right) \;\forall q \in {\varPi }_{h}.\]
\[\| w\|_{V} \leq \frac{1}{\beta }\| \mathcal{D}\left( {u - v}\right) \|_{\varPi } \leq \frac{C}{\beta }\| u - v\|_{V}.\]

\[{\begin{Vmatrix}u - {u}_{h}\end{Vmatrix}}_{V} + {\begin{Vmatrix}p - {p}_{h}\end{Vmatrix}}_{\varPi } \leq c\left( {\mathop{\inf }\limits_{{v \in {V}_{h}}}\| u - v\|_{V} + \mathop{\inf }\limits_{{q \in {\varPi }_{h}}}\| p - q\|_{\varPi }}\right) .\]
\[\left| {a\left( {u - {u}_{h},w}\right) }\right| \leq C\| w\|_{V}\mathop{\inf }\limits_{{q \in {\varPi }_{h}}}\| p - q\|_{\varPi }\]
\[{\begin{Vmatrix}u - {u}_{h}\end{Vmatrix}}_{V} \leq c\| u\|_{V}\]
